🧩 datetime.date#

기능설명
날짜 객체 생성datetime.date(연, 월, 일)
날짜 차이 계산day2 - day1 → timedelta
요일 확인.weekday(), .isoweekday() 사용
import datetime
d1 = datetime.date(2021, 12, 14)
d2 = datetime.date(2023, 4, 5)
print((d2 - d1).days)  # 날짜 차이

print(d1.weekday())     # 0=월, 6=일
print(d1.isoweekday())  # 1=월, 7=일

🧩 time 모듈#

함수설명
time()UTC 기준 초 단위 시간
localtime()현재 시각을 구조체로 반환
asctime() / ctime()사람이 읽기 좋은 형태로 변환
strftime()포맷 코드로 날짜 표현
sleep()코드 실행 일시 정지
import time
print(time.ctime())  # 현재 시간 출력
print(time.strftime('%Y-%m-%d %H:%M:%S'))  # 포맷 지정

for i in range(3):
    print(i)
    time.sleep(1)

🧩 itertools#

import itertools

# 순열
print(list(itertools.permutations(['a', 'b', 'c'], 2)))

# 조합
print(list(itertools.combinations(['a', 'b', 'c'], 2)))

# zip_longest
from itertools import zip_longest
print(list(zip_longest(['A', 'B'], [1], fillvalue='없음')))

🧩 functools.reduce#

from functools import reduce

nums = [1, 2, 3, 4]
total = reduce(lambda x, y: x + y, nums)
print(total)  # 10

🧩 operator.itemgetter#

🧩 sympy#

import sympy
from fractions import Fraction

x = sympy.symbols("x")
eq = sympy.Eq(x * Fraction(2, 5), 1760)
sol = sympy.solve(eq)
print(sol[0])  # 4400

# 연립방정식
x, y = sympy.symbols("x y")
eq1 = sympy.Eq(x + y, 10)
eq2 = sympy.Eq(x - y, 4)
print(sympy.solve([eq1, eq2]))  # {x: 7, y: 3}
